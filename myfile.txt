import nltk
from nltk.corpus import brown
from collections import Counter, defaultdict

# Download datasets
nltk.download('brown')
nltk.download('universal_tagset')

# Load corpus
corpus = brown.tagged_words(tagset='universal')[:5000]

# Count tags and word-tag pairs
tag_count = Counter()
word_tag_count = defaultdict(Counter)

for word, tag in corpus:
    word = word.lower()
    tag_count[tag] += 1
    word_tag_count[word][tag] += 1


def simple_hmm(sentence):

    words = sentence.lower().split()

    for word in words:

        print("\nWord:", word)
        print("Tag\tTran_Prob\tEmiss_Prob\tScore")

        best_tag = ""
        max_score = 0

        # MISTAKE 1:
        # You used tag without loop
        # Need loop for all tags

        for tag in tag_count:

            # Transition Probability
            tran_prob = tag_count[tag] / sum(tag_count.values())

            # Emission Probability
            emiss_prob = word_tag_count[word][tag] / tag_count[tag]

            # MISTAKE 2:
            # You used trans_prob instead of tran_prob

            score = tran_prob * emiss_prob

            print(f"{tag}\t{tran_prob:.4f}\t\t{emiss_prob:.4f}\t\t{score:.6f}")

            if score > max_score:
                max_score = score
                best_tag = tag

        print("BEST TAG:", best_tag)


sentence = input("Enter a sentence: ")
simple_hmm(sentence)

import nltk
from nltk.tokenize import word_tokenize
from nltk.stem import WordNetLemmatizer

# Downloads
nltk.download('punkt')
nltk.download('wordnet')
nltk.download('averaged_perceptron_tagger')

# Input
text = input("Enter your sentence: ")

# Tokenize words
words = word_tokenize(text)

# POS tagging
tags = nltk.pos_tag(words)

# Lemmatizer
lemmatizer = WordNetLemmatizer()

# Prefixes and suffixes
prefixes = ['un', 're', 'dis']
suffixes = ['ing', 'ed', 'er', 'tion']

# Heading
print("\nWord\t\tCategory\tPrefix\tRoot\t\tSuffix")
print("-" * 60)

# Process each word
for word, tag in tags:

    # Root word
    root = lemmatizer.lemmatize(word.lower())

    # Find prefix
    prefix = "None"

    for p in prefixes:
        if word.lower().startswith(p):
            prefix = p

    # Find suffix
    suffix = "None"

    for s in suffixes:
        if word.lower().endswith(s):
            suffix = s

    print(word, "\t\t", tag, "\t\t", prefix, "\t", root, "\t\t", suffix)

DFA

import java.util.Scanner;

public class DFA {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.print("Enter string: ");

        String str = sc.nextLine();


        int state = 0;

        // Check each character
        for(char ch : str.toCharArray()) {


            // Move from state 0 to 1 on 'a'
            if(state == 0 && ch == 'a') {

                state = 1;

            }


            // Move from state 1 to 2 on 'b'
            else if(state == 1 && ch == 'b') {

                state = 2;
            }


            // Otherwise go back to start state

            else {

                state = 0;
            }

        }

        // Final state check

        if(state == 2) {
            System.out.println("String Accepted");

        }
        else {
            System.out.println("String Rejected");

        }
    }

}

import nltk
from nltk.corpus import brown, treebank
from nltk.util import bigrams, trigrams
from collections import Counter


# Downloads
nltk.download('brown')
nltk.download('treebank')


# Clean Brown corpus words
brown_words = []


for word in brown.words():


    if word.isalpha():


        brown_words.append(word.lower())

# Clean WSJ corpus words

wsj_words = []


for word in treebank.words():

    if word.isalpha():


        wsj_words.append(word.lower())


# Input word

input_word = input("Enter a word: ").lower()



# ---------------- FUNCTION ----------------

def ngram_analysis(words, corpus_name):


    print("\nCorpus :", corpus_name)

    print("-" * 50)


    # ---------------- BIGRAMS ----------------
    bg_list = []


    for bg in bigrams(words):

        if bg[0] == input_word:


            bg_list.append(bg)


    bg_count = Counter(bg_list)

    print("\nBigrams")


    for bg, count in bg_count.most_common(5):


        probability = count / len(bg_list)

        print(bg, " Probability =", round(probability, 4))



    # ---------------- TRIGRAMS ----------------

    tg_list = []


    for tg in trigrams(words):


        if tg[0] == input_word:


            tg_list.append(tg)

    tg_count = Counter(tg_list)


    print("\nTrigrams")

    for tg, count in tg_count.most_common(5):


        probability = count / len(tg_list)

        print(tg, " Probability =", round(probability, 4))



# Brown Corpus
ngram_analysis(brown_words, "Brown")


# WSJ Corpus
ngram_analysis(wsj_words, "WSJ")

VITERBI

import nltk
from nltk.corpus import brown

from collections import defaultdict, Counter


# Downloads
nltk.download('brown')
nltk.download('universal_tagset')

nltk.download('punkt')


# Load tagged sentences

corpus = brown.tagged_sents(tagset='universal')[:2000]


# ---------------- TRAINING ----------------
transition = defaultdict(Counter)
emission = defaultdict(Counter)
tag_count = Counter()


for sentence in corpus:

    prev_tag = "START"


    for word, tag in sentence:

        word = word.lower()


        transition[prev_tag][tag] += 1
        emission[tag][word] += 1

        tag_count[tag] += 1


        prev_tag = tag


# ---------------- VITERBI FUNCTION ----------------

def viterbi(words):

    prev_tag = "START"

    result = []


    for word in words:


        best_tag = ""
        best_score = 0


        for tag in emission:


            # Transition probability
            trans_prob = transition[prev_tag][tag] + 1


            # Emission probability
            emit_prob = emission[tag][word.lower()] + 1

            # Viterbi score
            score = trans_prob * emit_prob


            if score > best_score:

                best_score = score
                best_tag = tag

        result.append((word, best_tag))

        prev_tag = best_tag

    return result

# ---------------- INPUT ----------------
sentence = input("Enter a sentence: ")

words = nltk.word_tokenize(sentence)

# ---------------- OUTPUT ----------------
tags = viterbi(words)

print("\nWord\tPOS Tag")
print("-" * 25)

for word, tag in tags:

    print(word, "\t", tag)
